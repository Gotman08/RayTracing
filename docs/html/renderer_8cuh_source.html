41__device__ inline Color ray_color(
42 const Ray& initial_ray,
43 const BVH& bvh,
44 int max_depth,
45 const RenderConfig& config,
46 curandState* rand_state
47) {
48 Color accumulated(0, 0, 0);
49 Color attenuation(1, 1, 1);
50 Ray current_ray = initial_ray;
51
52 for (int depth = 0; depth < max_depth; depth++) {
53 HitRecord rec;
54
55 if (bvh.hit(current_ray, Interval(0.001f, INFINITY_F), rec)) {
56 Ray scattered;
57 Color scatter_attenuation;
58
59 if (scatter(*rec.mat, current_ray, rec, scatter_attenuation, scattered, rand_state)) {
60 attenuation = attenuation * scatter_attenuation;
61 current_ray = scattered;
62 } else {
63 break;
64 }
65 } else {
66 if (config.use_sky) {
67 accumulated += attenuation * config.sky.get_color(current_ray.direction());
68 } else {
69 accumulated += attenuation * config.background;
70 }
71 break;
72 }
73 }
74
75 return accumulated;
76}

98 int i = blockIdx.x * blockDim.x + threadIdx.x;
99 int j = blockIdx.y * blockDim.y + threadIdx.y;
100
101 if (i >= config.width || j >= config.height) return;
102
103 int pixel_index = j * config.width + i;
104 curandState* local_rand_state = &rand_states[pixel_index];
105
106 Color pixel_color(0, 0, 0);
107
108 for (int s = 0; s < config.samples_per_pixel; s++) {
109 Ray r = camera.get_ray(i, j, local_rand_state);
110 pixel_color += ray_color(r, bvh, config.max_depth, config, local_rand_state);
111 }
112
113 pixel_color = pixel_color / static_cast<float>(config.samples_per_pixel);
114
115 pixel_color = apply_tone_mapping(pixel_color);
116 pixel_color = gamma_correct(pixel_color);
117
118 pixel_color.x = fminf(1.0f, fmaxf(0.0f, pixel_color.x));
119 pixel_color.y = fminf(1.0f, fmaxf(0.0f, pixel_color.y));
120 pixel_color.z = fminf(1.0f, fmaxf(0.0f, pixel_color.z));
121
122 frame_buffer[pixel_index] = pixel_color;
123}

135__device__ __forceinline__ unsigned int wang_hash(unsigned int seed) {
136 seed = (seed ^ 61) ^ (seed >> 16);
137 seed *= 9;
138 seed = seed ^ (seed >> 4);
139 seed *= 0x27d4eb2d;
140 seed = seed ^ (seed >> 15);
141 return seed;
142}
