131 int build_recursive(int start, int end) {
132 int node_idx = static_cast<int>(nodes.size());
133 nodes.emplace_back();
134
135 AABB bounds = primitives[start].bbox;
136 for (int i = start + 1; i < end; i++) {
137 bounds = AABB(bounds, primitives[i].bbox);
138 }
139 nodes[node_idx].bounds = bounds;
140
141 int count = end - start;
142
143 if (count == 1) {
144 nodes[node_idx].is_leaf = true;
145 nodes[node_idx].primitive_idx = start;
146 return node_idx;
147 }
148
149 int axis = bounds.longest_axis();
150
151 std::sort(primitives.begin() + start, primitives.begin() + end,
152 [axis](const HittableObject& a, const HittableObject& b) {
153 return a.bbox.centroid()[axis] < b.bbox.centroid()[axis];
154 });
155
156 int mid = start + count / 2;
157
158 nodes[node_idx].left = build_recursive(start, mid);
159 nodes[node_idx].right = build_recursive(mid, end);
160 nodes[node_idx].is_leaf = false;
161
