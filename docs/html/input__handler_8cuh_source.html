56 void update_mouse_delta() {
57 if (first_mouse) {
58 last_mouse_x = mouse_x;
59 last_mouse_y = mouse_y;
60 first_mouse = false;
61 }
62 mouse_delta_x = mouse_x - last_mouse_x;
63 mouse_delta_y = last_mouse_y - mouse_y;
64 last_mouse_x = mouse_x;
