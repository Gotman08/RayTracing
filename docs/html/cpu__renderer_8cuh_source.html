44inline Color ray_color_cpu(
45 const Ray& initial_ray,
46 const BVH& bvh,
47 int max_depth,
48 const RenderConfig& config,
49 CPURandom& rng
50) {
51 Color accumulated(0, 0, 0);
52 Color attenuation(1, 1, 1);
53 Ray current_ray = initial_ray;
54
55 for (int depth = 0; depth < max_depth; depth++) {
56 HitRecord rec;
57
58 if (bvh.hit(current_ray, Interval(0.001f, 1e30f), rec)) {
59 Ray scattered;
60 Color scatter_attenuation;
61
62 if (scatter_cpu(*rec.mat, current_ray, rec, scatter_attenuation, scattered, rng)) {
63 attenuation = attenuation * scatter_attenuation;
64 current_ray = scattered;
65 } else {
66 break;
67 }
68 } else {
69 if (config.use_sky) {
70 accumulated += attenuation * config.sky.get_color(current_ray.direction());
71 } else {
72 accumulated += attenuation * config.background;
73 }
74 break;
75 }
76 }
77
78 return accumulated;
79}

99) {
100 const int width = config.width;
101 const int height = config.height;
102 const int samples = config.samples_per_pixel;
103 const int max_depth = config.max_depth;
104
105 #pragma omp parallel for schedule(dynamic, 1) collapse(1)
106 for (int j = 0; j < height; j++) {
107 CPURandom rng(j * 1337 + 42);
108
109 for (int i = 0; i < width; i++) {
110 Color pixel_color(0, 0, 0);
111
112 for (int s = 0; s < samples; s++) {
113 Ray r = camera.get_ray_cpu(i, j, rng);
114 pixel_color += ray_color_cpu(r, bvh, max_depth, config, rng);
115 }
116
117 pixel_color = pixel_color / static_cast<float>(samples);
118
119 pixel_color = apply_tone_mapping(pixel_color);
120 pixel_color = gamma_correct(pixel_color);
121
122 pixel_color.x = fminf(1.0f, fmaxf(0.0f, pixel_color.x));
123 pixel_color.y = fminf(1.0f, fmaxf(0.0f, pixel_color.y));
124 pixel_color.z = fminf(1.0f, fmaxf(0.0f, pixel_color.z));
125
126 int pixel_index = j * width + i;
127 frame_buffer[pixel_index] = pixel_color;
128 }
129 }
