120private:
133 int build_recursive(int start, int end) {
134 int node_idx = static_cast<int>(nodes.size());
135 nodes.emplace_back();
136
137 AABB bounds = primitives[start].bbox;
138 for (int i = start + 1; i < end; i++) {
139 bounds = AABB(bounds, primitives[i].bbox);
140 }
141 nodes[node_idx].bounds = bounds;
142
143 int count = end - start;
144
145 if (count == 1) {
146 nodes[node_idx].is_leaf = true;
147 nodes[node_idx].primitive_idx = start;
148 return node_idx;
149 }
150
151 int axis = bounds.longest_axis();
152
153 std::sort(primitives.begin() + start, primitives.begin() + end,
154 [axis](const HittableObject& a, const HittableObject& b) {
155 return a.bbox.centroid()[axis] < b.bbox.centroid()[axis];
156 });
157
158 int mid = start + count / 2;
159
160 nodes[node_idx].left = build_recursive(start, mid);
161 nodes[node_idx].right = build_recursive(mid, end);
162 nodes[node_idx].is_leaf = false;
163
164 return node_idx;
165 }
